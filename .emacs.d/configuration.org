** Evil-Leader
#+BEGIN_SRC emacs-lisp
  (require 'evil-leader)
  (global-evil-leader-mode)
  (evil-leader/set-leader ";")
  (evil-leader/set-key
    "a" 'org-agenda
    "b" 'switch-to-buffer
    "c" 'org-capture
    "k" 'kill-buffer
    "l" 'org-insert-link
    "0" 'delete-window
    "'" 'split-window-horizontally
    "\"" 'split-window-vertically
    "#" 'linum-mode)
#+END_SRC
